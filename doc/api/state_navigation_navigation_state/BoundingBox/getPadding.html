BoundingPadding getPadding(double screenWidth, double screenHeight) {
  final double aspectRatio = width / height;
  final double screenAspectRatio = screenWidth / screenHeight;

  if (aspectRatio > screenAspectRatio) {
    // Route is wider than screen
    final double verticalPadding =
        (screenHeight - (screenWidth / aspectRatio)) / 2;
    return BoundingPadding(
      top: verticalPadding,
      bottom: verticalPadding,
      left: 0,
      right: 0,
    );
  } else {
    // Route is taller than screen
    final double horizontalPadding =
        (screenWidth - (screenHeight * aspectRatio)) / 2;
    return BoundingPadding(
      top: 0,
      bottom: 0,
      left: horizontalPadding,
      right: horizontalPadding,
    );
  }
}
