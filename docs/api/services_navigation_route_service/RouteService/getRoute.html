Future<Map<String, dynamic>> getRoute(Position start, Position destination) async {
  final url = Uri.https(_baseUrl, '$_routePath/${start.lng},${start.lat};${destination.lng},${destination.lat}', {
    'alternatives': 'true',
    'geometries': 'geojson',
    'overview': 'full',
    'steps': 'true',
    'access_token': accessToken,
    'annotations': 'maxspeed,congestion,closure'
  });

  final response = await http.get(url);
  if (response.statusCode == 200) {
    return json.decode(response.body);
  }
  throw Exception('Failed to get route: ${response.statusCode}');
}
