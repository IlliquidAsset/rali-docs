Future<void> displayRoute(MapboxMap map, Map<String, dynamic> routeData, {
  bool showAlternatives = false,
  RouteEmphasis emphasis = RouteEmphasis.balanced
}) async {
  await clearRoute(map);

  final routes = routeData['routes'] as List;
  if (routes.isEmpty) throw Exception('No routes available');

  // Add main route
  await _addRouteToMap(map, routes[0], emphasis);

  // Add alternative routes if requested
  if (showAlternatives) {
    for (var i = 1; i < routes.length; i++) {
      await _addRouteToMap(map, routes[i], emphasis,
        isAlternative: true,
        layerId: 'route-alt-$i'
      );
    }
  }

  // Calculate optimal view that includes all visible routes
  final allPoints = _getAllRoutePoints(routes, showAlternatives);
  final bounds = _calculateRouteBounds(allPoints);
  await map.flyTo(bounds, MapAnimationOptions(duration: 2000));
}
