factory RouteDetails.fromMapboxRoute(Map<String, dynamic> routeData) {
  final route = routeData['routes'][0];
  final geometry = route['geometry'];
  final coordinates = geometry['coordinates'] as List;
  final leg = route['legs'][0];

  // Parse steps
  final steps = (leg['steps'] as List)
      .map((step) => NavigationStep.fromMapboxStep(step))
      .toList();

  // Calculate bounds
  double minLng = coordinates[0][0], maxLng = coordinates[0][0];
  double minLat = coordinates[0][1], maxLat = coordinates[0][1];
  for (var coord in coordinates) {
    minLng = (coord[0] as double) < minLng ? coord[0] : minLng;
    maxLng = (coord[0] as double) > maxLng ? coord[0] : maxLng;
    minLat = (coord[1] as double) < minLat ? coord[1] : minLat;
    maxLat = (coord[1] as double) > maxLat ? coord[1] : maxLat;
  }

  final bounds = BoundingBox(
    southwest: Position(minLng, minLat),
    northeast: Position(maxLng, maxLat),
  );

  return RouteDetails(
    points: coordinates.map((coord) => Position(
      coord[0] as double,
      coord[1] as double,
    )).toList(),
    distance: leg['distance'] as double,
    duration: leg['duration'] as double,
    steps: steps,
    bounds: bounds,
  );
}
