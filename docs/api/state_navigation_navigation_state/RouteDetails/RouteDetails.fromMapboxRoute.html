factory RouteDetails.fromMapboxRoute(Map<String, dynamic> route) {
  final List<dynamic> coordinates = route['geometry']['coordinates'];
  final List<Position> points = coordinates
      .map((coord) => Position(coord[0] as double, coord[1] as double))
      .toList();

  final legs = route['legs'] as List<dynamic>;
  final steps = legs
      .expand((leg) => (leg['steps'] as List)
          .map((step) => NavigationStep.fromMapboxStep(step)))
      .toList();

  final bbox = route['bbox'] as List<dynamic>;
  final bounds = BoundingBox(
    southwest: Position(bbox[0] as double, bbox[1] as double),
    northeast: Position(bbox[2] as double, bbox[3] as double),
  );

  return RouteDetails(
    points: points,
    distance: (route['distance'] as num).toDouble(),
    duration: (route['duration'] as num).toDouble(),
    steps: steps,
    bounds: bounds,
  );
}
