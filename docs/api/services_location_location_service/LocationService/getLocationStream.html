Stream<Position> getLocationStream()

Provides a stream of location updates

Implementation

Stream<geo.Position> getLocationStream() {
  // Initialize stream if not already created
  _locationStreamController ??= StreamController<geo.Position>.broadcast();

  // Start listening to position updates
  _positionSubscription ??= geo.Geolocator.getPositionStream(
    locationSettings: geo.LocationSettings(
      accuracy: _locationAccuracy,
      distanceFilter: _minimumDistance.toInt(), // Minimum distance change to trigger update (meters)
      timeLimit: _updateInterval,
    ),
  ).listen(
    (geo.Position position) {
      _lastKnownLocation = position;
      _locationStreamController?.add(position);
    },
    onError: (error) {
      print('Location stream error: $error');
      _locationStreamController?.addError(error);
    },
    cancelOnError: false,
  );

  return _locationStreamController!.stream;
}
